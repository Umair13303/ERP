function processPayment(btnElement) {
    if (!validater()) return;

    var tr = $(btnElement).closest('tr');
    var rowData = invoiceTable.row(tr).data();
    var guID = rowData.guID;
    var invoiceId = rowData.invoiceId;

    // VALIDATION 1: Invoice ID must exist
    if (!invoiceId || invoiceId === 0 || invoiceId === '') {
        toastr.error("Invalid invoice selection. Please select a valid invoice.");
        return;
    }

    // VALIDATION 2: Payment amount must be > 0
    var receiptAmount = parseFloat($('#TextBoxReceiptAmount_' + guID).val());
    if (!receiptAmount || receiptAmount <= 0) {
        toastr.warning("Please enter a valid payment amount (greater than 0).");
        return;
    }


    // VALIDATION 3: Payment must not exceed due amount
    var dueAmount = parseFloat(rowData.dueAmount) || 0;
    if (receiptAmount > dueAmount) {
        toastr.error(
            "Payment amount (PKR " + receiptAmount.toFixed(2) + ") exceeds due amount (PKR " + dueAmount.toFixed(2) + "). " +
            "Please correct the payment amount."
        );
        return;
    }

    // If all validations pass, proceed with AJAX
    var jsonData = {
        OperationType: $("#OperationType").val(),
        LocationId: $("#DropDownListLocation :selected").val(),
        TransactionDate: $("#TextBoxTransactionDate").val(),
        CustomerId: $("#DropDownListCustomer :selected").val(),
        PaymentMethodId: $("#DropDownListPaymentMethod :selected").val(),
        Reference: $("#TextBoxReference").val(),
        InvoiceId: invoiceId,
        ReceiptAmount: receiptAmount,
        Description: $('#TextBoxDescription_' + guID).val(),
        PaymentTypeId: $("#PaymentTypeId").val(),
    };

    $.ajax({
        url: window.basePath + "AccountNfinance/AFInvoiceReceiptManagement/createUpdateInvoiceReceipt",
        type: "POST",
        data: JSON.stringify(jsonData),
        contentType: "application/json; charset=utf-8",
        dataType: "json",
        beforeSend: function () {
            initLoading();
            $(btnElement).prop("disabled", true);
        },
        success: function (response) {
            if (response.isSuccess) {
                toastr.success(response.message);
                $("#AFInvoiceReceipt").removeClass("was-validated");
                getInvoiceList($("#DropDownListCustomer :selected").val());
                clearInputFields();
            } else {
                toastr.error(response.message);  // Shows backend validation errors
            }
        },
        error: function (xhr) {
            toastr.error("System Error: " + (xhr.responseJSON?.message || xhr.statusText));
        },
        complete: function () {
            stopLoading();
            $(btnElement).prop("disabled", false);
        }
    });
}
